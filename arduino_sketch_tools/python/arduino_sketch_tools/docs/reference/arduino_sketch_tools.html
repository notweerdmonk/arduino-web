110    def _compute_sketch_checksum(self, sketch_dir: str) -> str:
111        """Compute a SHA-256 checksum over all source files in a sketch directory.
112
113        Args:
114            sketch_dir: Path to the sketch directory.
115
116        Returns:
117            Hex digest string, or empty string if no source files found.
118        """
119        if not os.path.isdir(sketch_dir):
120            return ""
121        import hashlib
122
123        hasher = hashlib.sha256()
124        file_paths = []
125        for root, _dirs, files in os.walk(sketch_dir):
126            for f in files:
127                if f.endswith((".ino", ".cpp", ".h", ".hpp", ".c")):
128                    file_paths.append(os.path.join(root, f))
129        file_paths.sort()
130        for fp in file_paths:
131            try:
132                with open(fp, "rb") as fh:
133                    while True:
134                        chunk = fh.read(65536)
135                        if not chunk:
136                            break
137                        hasher.update(chunk)
138            except OSError:
139                continue
140        if not file_paths:
141            return ""
142        return hasher.hexdigest()

144    def _get_sketch_mtime(self, sketch_path: str) -> float | None:
145        """Return the latest modification time among source files in a sketch.
146
147        Args:
148            sketch_path: Path to the sketch directory.
149
150        Returns:
151            Latest mtime as float, or None if the directory is empty or missing.
152        """
153        if not os.path.isdir(sketch_path):
154            return None
155        max_mtime: float | None = None
156        for root, _dirs, files in os.walk(sketch_path):
157            for f in files:
158                if f.endswith((".ino", ".cpp", ".h", ".hpp", ".c")):
159                    try:
160                        mtime = os.path.getmtime(os.path.join(root, f))
161                        if max_mtime is None or mtime > max_mtime:
162                            max_mtime = mtime
163                    except OSError:
164                        continue
165        return max_mtime

191    def _on_compile_resp(self, msg: dict) -> None:
192        """Handle a compile response message from pubsub.
193
194        Progress messages are broadcast as WebSocket HTML; final results
195        are stored in _compile_results and update sketch metadata.
196        """
197        topic = msg.get("topic", "")
198        if not topic.startswith("resp::compile::"):
199            return
200        rest = topic[len("resp::compile::") :]
201        is_progress = rest.endswith("::progress")
202        if is_progress:
203            port = rest[: -len("::progress")]
204            data = msg.get("data", {})
205            out = data.get("output", "")
206            err = data.get("error", "")
207            percent = data.get("percent", 0.0)
208            port_safe = port.replace("/", "_")
209            with self._compile_last_pct_lock:
210                last = self._compile_last_pct.get(port_safe, -1.0)
211                if percent != last:
212                    pct_int = int(round(percent))
213                    self._compile_last_pct[port_safe] = percent
214                    bar = f'<progress id="compile-progress-{port_safe}" value="{pct_int}" max="100" hx-swap-oob="true"></progress>'
215                    self._broadcast_ws(bar)
216            if out or err:
217                text = (
218                    (out + err)
219                    .replace("&", "&amp;")
220                    .replace("<", "&lt;")
221                    .replace(">", "&gt;")
222                )
223                pct_prefix = f"[{pct_int}%] " if percent >= 0 else ""
224                html = f'<span hx-swap-oob="beforeend:#compile-output-{port_safe}"><div class="compile-line" data-port="{port}">{pct_prefix}{text}</div></span>'
225                self._broadcast_ws(html)
226        else:
227            port = rest
228            with self._compile_results_lock:
229                self._compile_results[port] = msg
230            if msg.get("status") == "ok":
231                sketch_path = msg.get("data", {}).get("sketch_path", "")
232                if sketch_path:
233                    with self._last_compiled_sketch_lock:
234                        self._last_compiled_sketch[port] = sketch_path
235                    with self._last_compile_mtime_lock:
236                        self._last_compile_mtime[port] = self._get_sketch_mtime(
237                            sketch_path
238                        )
239
240    def _on_upload_resp(self, msg: dict) -> None:
241        """Handle an upload response message from pubsub.
242
243        Progress messages are broadcast as WebSocket HTML; final results
244        are stored in _upload_results and trigger deploy recording.
245        """
246        topic = msg.get("topic", "")
247        if not topic.startswith("resp::upload::"):
248            return
249        rest = topic[len("resp::upload::") :]
250        is_progress = rest.endswith("::progress")
251        if is_progress:
252            port = rest[: -len("::progress")]
253            data = msg.get("data", {})
254            out = data.get("output", "")
255            err = data.get("error", "")
256            if out or err:
257                text = (
258                    (out + err)
259                    .replace("&", "&amp;")
260                    .replace("<", "&lt;")
261                    .replace(">", "&gt;")
262                )
263                port_safe = port.replace("/", "_")
264                html = f'<span hx-swap-oob="beforeend:#upload-output-{port_safe}"><div class="upload-line" data-port="{port}">{text}</div></span>'
265                self._broadcast_ws(html)
266        else:
267            port = rest
268            with self._upload_results_lock:
269                self._upload_results[port] = msg
270            if msg.get("status") == "ok":
271                with self._upload_meta_lock:
272                    sketch_path = self._upload_meta.get(port, {}).get("sketch", "")
273                if sketch_path:
274                    self._record_deploy(port, sketch_path)

class SketchRegistry: 
 29class SketchRegistry:
 30    """Thread-safe registry mapping hardware IDs to sketch paths.
 31
 32    Operates on a shared upload registry dict (passed at init) whose
 33    structure is::
 34
 35        {(ip, ua): {sketch_name: [version_dict, ...]}}
 36
 37    Each version dict may contain a ``"hardware_ids"`` list that maps
 38    this board to a specific sketch version.
 39    """
 40
 41    def __init__(self, registry: dict, lock: threading.Lock) -> None:
 42        """Init the registry with a shared upload dict and locks.
 43
 44        Args:
 45            registry: Shared upload registry dict.
 46            lock: External lock guarding the registry.
 47        """
 48        self._registry = registry
 49        self._lock = lock
 50        self._op_lock = threading.Lock()
 51
 52    def get_assignment(self, hardware_id: str) -> Optional[str]:
 53        """Return the sketch directory assigned to a hardware ID, or None.
 54
 55        Args:
 56            hardware_id: The board hardware ID to look up.
 57        """
 58        if not hardware_id:
 59            return None
 60        with self._op_lock:
 61            with self._lock:
 62                for entry in self._registry.values():
 63                    for versions in entry.values():
 64                        for v in versions:
 65                            if hardware_id in v.get(
 66                                "hardware_ids", []
 67                            ) and os.path.isdir(v["path"]):
 68                                return v["path"]
 69        return None
 70
 71    def set_assignment(self, hardware_id: str, sketch_dir: str) -> None:
 72        """Assign a sketch directory to a hardware ID.
 73
 74        Args:
 75            hardware_id: The board hardware ID to assign.
 76            sketch_dir: The sketch directory path to assign.
 77        """
 78        if not hardware_id:
 79            return
 80        with self._op_lock:
 81            with self._lock:
 82                for entry in self._registry.values():
 83                    for versions in entry.values():
 84                        for v in versions:
 85                            if v["path"] == sketch_dir:
 86                                if hardware_id not in v.get("hardware_ids", []):
 87                                    v.setdefault("hardware_ids", []).append(hardware_id)
 88                                return
 89
 90    def clear_assignment(self, hardware_id: str) -> None:
 91        """Remove the assignment for a given hardware ID.
 92
 93        Args:
 94            hardware_id: The board hardware ID to clear.
 95        """
 96        if not hardware_id:
 97            return
 98        with self._op_lock:
 99            with self._lock:
100                for entry in self._registry.values():
101                    for versions in entry.values():
102                        for v in versions:
103                            if hardware_id in v.get("hardware_ids", []):
104                                v["hardware_ids"].remove(hardware_id)
105                                return
106
107    def get_all_assignments(self) -> dict[str, str]:
108        """Return a copy of all hardware ID to sketch path mappings."""
109        result = {}
110        with self._op_lock:
111            with self._lock:
112                for entry in self._registry.values():
113                    for versions in entry.values():
114                        for v in versions:
115                            for hw_id in v.get("hardware_ids", []):
116                                if os.path.isdir(v["path"]):
117                                    result[hw_id] = v["path"]
118        return result
