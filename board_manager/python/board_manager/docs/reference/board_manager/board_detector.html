 43class BoardDetector:
 44    """Detects board connect/disconnect events.
 45
 46    Supports two modes:
 47    - ``"watch"`` (default): uses arduino-cli BoardListWatch streaming RPC.
 48    - ``"poll"``: legacy periodic polling via ``list_boards()``.
 49
 50    Runs a background thread that invokes a callback with ``"connected"`` or
 51    ``"disconnected"`` events. On connection failures, waits 2 seconds before
 52    retrying. If a ``daemon_manager`` is configured, attempts to restart the
 53    daemon before retrying.
 54    """
 55
 56    def __init__(
 57        self,
 58        callback: BoardEventCallback,
 59        daemon: str = "localhost:50051",
 60        poll_interval: float = DetectorDefaults.POLL_INTERVAL.value,
 61        list_timeout: float = DetectorDefaults.LIST_TIMEOUT.value,
 62        daemon_manager: Any = None,
 63        mode: str = "watch",
 64    ):
 65        """Initialize the board detector.
 66
 67        Args:
 68            callback: Called with (port, msg) on connect/disconnect events.
 69            daemon: Arduino CLI daemon address.
 70            poll_interval: Seconds between poll iterations.
 71            list_timeout: Timeout for list_boards gRPC call.
 72            daemon_manager: Optional DaemonManager for auto-recovery.
 73            mode: Detection mode — ``"watch"`` or ``"poll"``.
 74        """
 75        self._callback = callback
 76        self._daemon = daemon
 77        self._poll_interval = poll_interval
 78        self._list_timeout = list_timeout
 79        self._daemon_manager = daemon_manager
 80        self._mode = mode
 81        self._known_boards: dict[str, dict] = {}
 82        self._lock = threading.Lock()
 83        self._running = False
 84        self._thread: Optional[threading.Thread] = None
 85
 86    def start(self) -> None:
 87        """Start the detector background thread."""
 88        if self._running:
 89            return
 90        self._running = True
 91        if self._mode == "watch":
 92            self._thread = threading.Thread(target=self._run_watch, daemon=True)
 93            self._thread.start()
 94            logger.info("BoardDetector started (watch mode)")
 95        else:
 96            self._thread = threading.Thread(target=self._run, daemon=True)
 97            self._thread.start()
 98            logger.info(
 99                "BoardDetector started (poll mode, every %.1fs)", self._poll_interval
100            )
101
102    def stop(self) -> None:
103        """Stop the detector background thread."""
104        if not self._running:
105            return
106        self._running = False
107        logger.info("BoardDetector stopped")
108
109    def _connect_or_restart(self, silent: bool = False) -> Optional[ArduinoGrpcClient]:
110        """Try to connect/init; on failure restart daemon and retry once."""
111        client = ArduinoGrpcClient(daemon=self._daemon)
112        try:
113            client.connect()
114            client.init()
115            return client
116        except Exception as e:
117            if not silent:
118                logger.warning(
119                    "BoardDetector: failed to connect to daemon (%s), retrying...", e
120                )
121            client.disconnect()
122            if self._restart_daemon():
123                client = ArduinoGrpcClient(daemon=self._daemon)
124                try:
125                    client.connect()
126                    client.init()
127                    return client
128                except Exception as e2:
129                    logger.warning(
130                        "BoardDetector: still cannot connect after restart (%s)", e2
131                    )
132                    client.disconnect()
133            return None
134
135    def _restart_daemon(self) -> bool:
136        """Restart the arduino-cli daemon via daemon_manager if available.
137
138        Returns:
139            True if the daemon was successfully restarted.
140        """
141        if not self._daemon_manager:
142            return False
143        try:
144            ok = self._daemon_manager.ensure_alive()
145            if ok:
146                logger.info("BoardDetector: daemon restarted successfully")
147            else:
148                logger.warning("BoardDetector: daemon restart failed")
149            return ok
150        except Exception as e:
151            logger.warning("BoardDetector: failed to restart daemon: %s", e)
152            return False
153
154    def get_known_boards(self) -> dict[str, dict]:
155        """Return a snapshot of currently known boards."""
156        with self._lock:
157            return dict(self._known_boards)
158
159    def _run(self) -> None:
160        """Poll loop: continuously poll board list and detect changes."""
161        while self._running:
162            try:
163                ok = self._run_once()
164            except Exception as e:
165                logger.error("BoardDetector: unexpected error: %s", e)
166                ok = False
167            if not self._running:
168                break
169            if ok:
170                time.sleep(self._poll_interval)
171            else:
172                logger.info(
173                    "BoardDetector: poll failed, next attempt in 2.0s",
174                )
175                time.sleep(2.0)
176
177    def _run_once(self) -> bool:
178        """Perform a single poll cycle: connect, list boards, emit events.
179
180        Returns:
181            True if the cycle completed without connection errors.
182        """
183        client = self._connect_or_restart()
184        if client is None:
185            return False
186
187        try:
188            while self._running:
189                try:
190                    boards = client.list_boards(timeout=self._list_timeout)
191                except Exception as e:
192                    logger.warning(
193                        "BoardDetector: list_boards failed (%s), reconnecting...", e
194                    )
195                    client.disconnect()
196                    self._restart_daemon()
197                    return False
198
199                current = {}
200                for b in boards:
201                    key = b.port.address
202                    current[key] = {
203                        "port": b.port.address,
204                        "fqbn": b.fqbn,
205                        "name": b.name,
206                        "hardware_id": b.port.hardware_id,
207                    }
208
209                with self._lock:
210                    old = dict(self._known_boards)
211                    self._known_boards = current
212
213                for addr, info in current.items():
214                    if addr not in old:
215                        logger.info(
216                            "Board detected: %s at %s (%s)",
217                            info["name"],
218                            addr,
219                            info["fqbn"],
220                        )
221                        self._emit("connected", info)
222
223                for addr, info in old.items():
224                    if addr not in current:
225                        logger.info(
226                            "Board disconnected: %s at %s (%s)",
227                            info["name"],
228                            addr,
229                            info.get("fqbn", ""),
230                        )
231                        self._emit("disconnected", info)
232                time.sleep(self._poll_interval)
233        finally:
234            try:
235                client.disconnect()
236            except Exception:
237                pass
238
239        return True
240
241    def _run_watch(self) -> None:
242        """Stream board events via BoardListWatch."""
243        while self._running:
244            client = self._connect_or_restart(silent=False)
245            if client is None:
246                time.sleep(2.0)
247                continue
248            try:
249                for board in client.watch_boards(timeout=None):
250                    if not self._running:
251                        break
252                    addr = board.port.address
253                    info = {
254                        "port": addr,
255                        "fqbn": board.fqbn,
256                        "name": board.name,
257                        "hardware_id": board.port.hardware_id,
258                        "source": "watch",
259                    }
260                    with self._lock:
261                        if board.detected and addr not in self._known_boards:
262                            self._known_boards[addr] = info
263                            event = "connected"
264                            payload = info
265                        elif not board.detected and addr in self._known_boards:
266                            payload = self._known_boards.pop(addr)
267                            event = "disconnected"
268                        else:
269                            event = None
270                    if event:
271                        logger.info(
272                            "Board %s: %s at %s (%s)",
273                            event,
274                            info["name"],
275                            addr,
276                            info["fqbn"],
277                        )
278                        self._emit(event, payload)
279            except Exception as e:
280                logger.warning(
281                    "BoardDetector: watch stream error (%s), reconnecting...", e
282                )
283                try:
284                    client.disconnect()
285                except Exception:
286                    pass
287                time.sleep(2.0)
288
289    def _emit(self, event: str, info: dict) -> None:
290        """Emit a board event to the registered callback.
291
292        Args:
293            event: ``"connected"`` or ``"disconnected"``.
294            info: Board info dict.
295        """
296        msg = {
297            "type": "event",
298            "topic": f"board::{info['port']}::event",
299            "data": {
300                "event": event,
301                "port": info["port"],
302                "board": info.get("name", ""),
303                "fqbn": info.get("fqbn", ""),
304                "hardware_id": info.get("hardware_id", ""),
305            },
306        }
307        try:
308            self._callback(info["port"], msg)
309        except Exception as e:
310            logger.error("BoardDetector: callback error: %s", e)
