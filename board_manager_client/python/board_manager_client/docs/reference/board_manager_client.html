 69
 70        Args:

 93    def connect(self, retry: bool = False) -> None:
 94        """Connect to the BoardManagerService.
 95
 96        Args:
 97            retry: If True, retry with delays from CONNECT_RETRY_DELAYS on failure.
 98
 99        Raises:
100            ConnectionError: If all connection attempts fail.
101            OSError: On socket-level errors.
102        """
103        if retry:
104            last_error: Optional[Exception] = None
105            for attempt, delay in enumerate(ReconnectConfig.CONNECT_RETRY_DELAYS):
106                try:
107                    self._connect_once()
108                    logger.info(
109                        "Connected to BoardManagerService (mode: %s)",
110                        "UDS"
111                        if self.use_uds
112                        and self._sock
113                        and self._sock.family == socket.AF_UNIX
114                        else "TCP",
115                    )
116                    return
117                except (ConnectionError, OSError) as e:
118                    last_error = e
119                    logger.warning(
120                        "Connect attempt %d/%d failed: %s, retrying in %.1fs...",
121                        attempt + 1,
122                        len(ReconnectConfig.CONNECT_RETRY_DELAYS),
123                        e,
124                        delay,
125                    )
126                    time.sleep(delay)
127            logger.error("All connect attempts failed: %s", last_error)
128            raise last_error  # type: ignore[misc]
129        self._connect_once()
130        logger.info(
131            "Connected to BoardManagerService (mode: %s)",
132            "UDS"
133            if self.use_uds and self._sock and self._sock.family == socket.AF_UNIX
134            else "TCP",
135        )

137    def _connect_once(self) -> None:
138        """Perform a single connection attempt, including handshake and resubscribe."""
139        self._sock = self._create_socket()
140        self._send_handshake()
141        self._resubscribe()
142        self._reconnect_count = 0
143        if self.on_reconnect:
144            self.on_reconnect()
145
146    def _create_socket(self) -> socket.socket:
147        """Create and connect a socket (UDS preferred, TCP fallback).
148
149        Returns:
150            A connected socket object.
151
152        Raises:
153            ConnectionError: If all transport options fail.
154        """
155        if self.use_uds and os.path.exists(self.uds_path):
156            sock = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
157            try:
158                sock.connect(self.uds_path)
159            except ConnectionRefusedError:
160                logger.warning("Stale UDS socket at %s, cleaning up", self.uds_path)
161                try:
162                    os.unlink(self.uds_path)
163                except OSError:
164                    pass
165                sock.close()
166                if os.path.exists(self.uds_path):
167                    sock = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
168                    sock.connect(self.uds_path)
169                    self._reader = FrameReader("newline")
170                    return sock
171                logger.warning("UDS socket cleaned, falling back to TCP")
172            else:
173                self._reader = FrameReader("newline")
174                return sock
175        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
176        sock.connect((self.tcp_host, self.tcp_port))
177        self._reader = FrameReader("newline")
178        return sock
179
180    def _send_handshake(self) -> None:
181        """Send the initial handshake if using TCP transport."""
182        if not self.use_uds or not os.path.exists(self.uds_path):
183            self._sock.sendall(Handshake.NEWLINE.value)

191        """Disconnect from the service and close the socket."""

195                self._sock.close()
196            except OSError:
197                pass
198            self._sock = None

200    def subscribe(self, topic: str, handler: Optional[EventHandler] = None) -> None:
201        """Subscribe to a topic, optionally registering a handler.
202
203        Args:
204            topic: Topic pattern to subscribe to.
205            handler: Optional callback invoked on matching messages.
206        """
207        with self._lock:
208            is_new = topic not in self._subscriptions
209            self._subscriptions.add(topic)
210            if handler:
211                hlist = self._handlers.setdefault(topic, [])
212                if handler not in hlist:
213                    hlist.append(handler)
214        if self._sock and is_new:
215            self._send({"type": "subscribe", "topic": topic})
216
217    def unsubscribe(self, topic: str) -> None:
218        """Unsubscribe from a topic and remove its handlers.
219
220        Args:
221            topic: Topic pattern to unsubscribe from.
222        """
223        with self._lock:
224            self._subscriptions.discard(topic)
225            self._handlers.pop(topic, None)
226        if self._sock:
227            self._send({"type": "unsubscribe", "topic": topic})

229    def request_boards(self, timeout: float = 10.0) -> list[dict] | None:
230        """Request the list of boards and wait for the response.
231
232        Args:
233            timeout: Maximum seconds to wait for a response.
234
235        Returns:
236            List of board dicts, or None if the request timed out.
237        """
238        resp_topic = f"resp::list_boards::{id(self)}::{time.monotonic_ns()}"
239        event = threading.Event()
240        with self._resp_lock:
241            self._pending_responses[resp_topic] = (None, event)
242        self.subscribe(resp_topic, lambda msg: self._on_response(resp_topic, msg))
243        self.publish("sys::boards", {"method": "list_boards"}, reply_to=resp_topic)
244        if event.wait(timeout):
245            with self._resp_lock:
246                result, _ = self._pending_responses.pop(resp_topic, (None, None))
247            return result.get("data", {}).get("boards", []) if result else None
248        with self._resp_lock:
249            self._pending_responses.pop(resp_topic, None)
250        self.unsubscribe(resp_topic)
251        return None
252
253    def _on_response(self, topic: str, msg: dict) -> None:
254        """Store a response message and signal the waiting caller.
255
256        Args:
257            topic: Response topic.
258            msg: Response message dict.
259        """
260        with self._resp_lock:
261            entry = self._pending_responses.get(topic)
262        if entry is not None:
263            result, event = entry
264            if not event.is_set():
265                with self._resp_lock:
266                    self._pending_responses[topic] = (msg, event)
267                event.set()
268
269    def publish(self, topic: str, message: dict, reply_to: str = "") -> None:
270        """Publish a message on a topic.
271
272        Args:
273            topic: Topic to publish on.
274            message: Message payload dict.
275            reply_to: Optional response topic for subscribers.
276        """
277        self._send(
278            {
279                "type": "publish",
280                "topic": topic,
281                "body": message,
282                "reply_to": reply_to,
283            }
284        )

286    def _send(self, msg: dict) -> None:
287        """Send a framed message over the socket.
288
289        Args:
290            msg: Message dict to encode and send.
291        """
292        if not self._sock:
293            return
294        try:
295            self._sock.sendall(encode_and_frame(msg, "newline"))
296        except (OSError, ConnectionError) as e:
297            logger.warning("Send failed: %s", e)
298            if self._sock:
299                try:
300                    self._sock.close()
301                except OSError:
302                    pass
303                self._sock = None
304
305    def _resubscribe(self) -> None:
306        """Re-send all subscription requests after reconnection."""
307        with self._lock:
308            topics = list(self._subscriptions)
309        for topic in topics:
310            self._send({"type": "subscribe", "topic": topic})

312    def start_reader(self) -> None:
313        """Start the background reader thread to process incoming messages."""
314        self._running = True
315        self._reader_thread = threading.Thread(target=self._read_loop, daemon=True)
316        self._reader_thread.start()
317
318    def _read_loop(self) -> None:
319        """Continuously read and dispatch messages from the socket, reconnecting on error."""
320        while self._running:
321            if not self._sock:
322                logger.debug("_read_loop: no socket, reconnecting")
323                self._reconnect()
324                time.sleep(0.1)
325                continue
326            try:
327                sock = self._sock
328                ready, _, _ = select.select([sock], [], [], 1.0)
329                if not ready:
330                    continue
331                data = sock.recv(65536)
332            except (
333                OSError,
334                ConnectionError,
335                ValueError,
336                TypeError,
337                AttributeError,
338            ) as e:
339                if self._running:
340                    logger.warning("Connection lost (%s), reconnecting...", e)
341                    self._reconnect()
342                continue
343
344            if not data:
345                if self._running:
346                    logger.warning("Connection closed (empty recv), reconnecting...")
347                    self._reconnect()
348                continue
349
350            logger.debug("_read_loop: received %d bytes", len(data))
351            try:
352                self._reader.feed(data)
353                while True:
354                    msg = self._reader.read_one()
355                    if msg is None:
356                        break
357                    logger.debug(
358                        "_read_loop: dispatching msg with topic='%s'",
359                        msg.get("topic", ""),
360                    )
361                    self._dispatch(msg)
362            except Exception as e:
363                logger.error("_read_loop: error processing message (%s), skipping", e)
364
365    def _dispatch(self, msg: dict) -> None:
366        """Dispatch a message to all matching handlers.
367
368        Args:
369            msg: Message dict to dispatch.
370        """
371        topic = msg.get("topic", "")
372        with self._lock:
373            matched: list[EventHandler] = []
374            for pattern, hlist in self._handlers.items():
375                if pattern == "*" or _match(topic, pattern):
376                    logger.debug(
377                        "_dispatch: topic '%s' matched pattern '%s' (%d handlers)",
378                        topic,
379                        pattern,
380                        len(hlist),
381                    )
382                    matched.extend(hlist)
383        if not matched:
384            logger.debug(
385                "_dispatch: topic '%s' matched NO patterns (known patterns: %s)",
386                topic,
387                list(self._handlers.keys()),
388            )
389        for h in matched:
390            try:
391                h(msg)
392            except Exception as e:
393                logger.error("Handler error: %s", e)
394
395    def _reconnect(self) -> None:
396        """Close the current socket, wait, and attempt a new connection."""
397        self._reconnect_count += 1
398        if self._sock:
399            try:
400                self._sock.close()
401            except OSError:
402                pass
403            self._sock = None
404        time.sleep(ReconnectConfig.RECONNECT_DELAY)
405        try:
406            self.connect()
407        except (ConnectionError, OSError) as e:
408            logger.warning(
409                "Reconnect failed (%s), retrying in %.1fs...",
410                e,
411                ReconnectConfig.RECONNECT_DELAY,
412            )
