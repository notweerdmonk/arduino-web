 44    def __init__(self, daemon: str = DEFAULT_DAEMON):
 45        """Initialize the client.
 46
 47        Args:
 48            daemon: Host:port of the arduino-cli daemon (default: localhost:50051).
 49        """
 50        self.daemon = daemon
 51        self.channel: Optional[grpc.Channel] = None
 52        self.stub: Optional[rpc_pb2_grpc.ArduinoCoreServiceStub] = None
 53        self._instance: Optional[common_pb2.Instance] = None

 55    def connect(self) -> None:
 56        """Connect to the arduino-cli daemon.
 57
 58        Creates an insecure gRPC channel to the configured daemon address.
 59        The channel is reused until disconnect() is called.
 60
 61        Raises:
 62            ConnectionError: If the daemon is unreachable or connection fails.
 63        """
 64        try:
 65            self.channel = grpc.insecure_channel(self.daemon)
 66            self.stub = rpc_pb2_grpc.ArduinoCoreServiceStub(self.channel)
 67        except Exception as e:
 68            raise exceptions.ConnectionError(f"Failed to connect to {self.daemon}: {e}")
 69
 70    def destroy(self) -> None:
 71        """Destroy the Arduino core instance on the daemon.
 72
 73        Each Create() should be paired with a Destroy() to free daemon-side
 74        resources. Called automatically by disconnect(). Failures are silently
 75        ignored (daemon may already be gone).
 76        """
 77        if self.stub and self._instance:
 78            try:
 79                self.stub.Destroy(rpc_pb2.DestroyRequest(instance=self._instance))
 80            except grpc.RpcError:
 81                pass
 82        self._instance = None
 83
 84    def disconnect(self) -> None:
 85        """Disconnect from the daemon and free resources.
 86
 87        Calls destroy() to release the daemon-side instance, then closes the
 88        gRPC channel. Safe to call multiple times.
 89        """
 90        self.destroy()
 91        if self.channel:
 92            self.channel.close()
 93            self.channel = None
 94            self.stub = None
 95
 96    def __enter__(self):
 97        """Enter context manager, connecting to the daemon."""
 98        self.connect()
 99        return self

101    def __exit__(self, exc_type, exc_val, exc_tb):
102        """Exit context manager, disconnecting from the daemon."""
103        self.disconnect()

105    def _ensure_connected(self):
106        """Raise ConnectionError if not connected to the daemon."""
107        if not self.stub:
108            raise exceptions.ConnectionError("Not connected. Call connect() first.")

110    def _ensure_instance(self):
111        """Return initialized instance, lazily initializing if needed."""
112        if not self._instance:
113            self._instance = self.init()

116    def init(self, sketch_path: Optional[str] = None) -> common_pb2.Instance:
117        """Initialize Arduino core instance
118
119        Creates a new instance via Create RPC, then initializes it via Init RPC.
120        The Init streaming response is fully consumed to ensure platform data loads.
121
122        Args:
123            sketch_path: Optional sketch path for Init context
124
125        Returns:
126            Initialized Instance object
127
128        Raises:
129            InitializationError: If initialization fails
130        """
131        self._ensure_connected()
132        try:
133            create_resp = self.stub.Create(rpc_pb2.CreateRequest())
134            inst = create_resp.instance
135
136            init_req = rpc_pb2.InitRequest(instance=inst)
137            if sketch_path:
138                init_req.sketch_path = sketch_path
139
140            for _ in self.stub.Init(init_req):
141                pass
142
143            self._instance = inst
144            return self._instance
145
146        except grpc.RpcError as e:
147            raise exceptions.InitializationError(f"Init RPC failed: {e.details()}")

160    def list_boards(self, timeout: int = 5) -> List[Board]:
161        """List currently connected boards
162
163        Args:
164            timeout: Seconds to probe serial ports (default 5). The daemon
165                     needs time to enumerate USB/serial devices.
166
167        Returns:
168            List of Board objects from detected ports
169        """
170        self._ensure_connected()
171        instance = self._ensure_instance()
172
173        try:
174            request = board_pb2.BoardListRequest(
175                instance=instance, timeout=int(timeout)
176            )
177            response = self.stub.BoardList(request)
178
179            boards = []
180            for dp in response.ports:
181                port_addr = dp.port.address if dp.port else ""
182                port_proto = dp.port.protocol if dp.port else ""
183                for mb in dp.matching_boards:
184                    board = Board(
185                        port=Port(address=port_addr, protocol=port_proto),
186                        fqbn=mb.fqbn,
187                        name=mb.name,
188                        detected=True,
189                    )
190                    boards.append(board)
191            return boards
192
193        except grpc.RpcError as e:
194            raise exceptions.BoardError(f"BoardList RPC failed: {e.details()}")

223    def watch_boards(
224        self,
225        callback: Optional[Callable[[Board], None]] = None,
226        timeout: Optional[float] = None,
227    ) -> Iterator[Board]:
228        """Stream board connect/disconnect events.
229
230        Yields Board objects as boards are connected or disconnected.
231        Blocks until an event arrives or the optional timeout expires.
232
233        Args:
234            callback: Optional function to call for each board event (in
235                      addition to yielding).
236            timeout: Maximum seconds to wait for the first event. When the
237                     deadline passes, gRPC raises DEADLINE_EXCEEDED which is
238                     handled gracefully (stops iteration) rather than raised
239                     as an error.
240
241        Yields:
242            Board objects representing the detected port and matching board.
243
244        Raises:
245            BoardError: If the BoardListWatch RPC fails with an error other
246                        than DEADLINE_EXCEEDED.
247        """
248        self._ensure_connected()
249        instance = self._ensure_instance()
250
251        try:
252            request = board_pb2.BoardListWatchRequest(instance=instance)
253            kwargs = {"request": request}
254            if timeout is not None:
255                kwargs["timeout"] = timeout
256            for response in self.stub.BoardListWatch(**kwargs):
257                if response.error:
258                    raise exceptions.BoardError(f"Board watch error: {response.error}")
259
260                if response.HasField("port") and response.port.port.address:
261                    fqbn = ""
262                    name = ""
263                    if response.port.matching_boards:
264                        fb = response.port.matching_boards[0]
265                        fqbn = fb.fqbn
266                        name = fb.name
267                    board = Board(
268                        port=Port(
269                            address=response.port.port.address,
270                            protocol=response.port.port.protocol,
271                            protocol_label=response.port.port.protocol_label,
272                            label=response.port.port.label,
273                        ),
274                        fqbn=fqbn,
275                        name=name,
276                        detected=response.event_type == "add",
277                    )
278                    if callback:
279                        callback(board)
280                    yield board
281
282        except grpc.RpcError as e:
283            if e.code() == grpc.StatusCode.DEADLINE_EXCEEDED:
284                return
285            raise exceptions.BoardError(f"BoardListWatch RPC failed: {e.details()}")

287    def compile_stream(
288        self,
289        sketch_path: str,
290        fqbn: str,
291        verbose: bool = False,
292        quiet: bool = False,
293    ) -> Iterator[tuple[str, str, bool, float]]:
294        """Stream compile output as it arrives from the gRPC stream.
295
296        Yields ``(out, err, done, percent)`` tuples for each response from the
297        Compile RPC.  *out* and *err* are decoded text chunks (may be empty).
298        *done* is True on the final response that contains the result.
299        *percent* is a 0.0–100.0 progress percentage from TaskProgress.
300
301        Args:
302            sketch_path: Path to the sketch directory containing the .ino file.
303            fqbn: Fully Qualified Board Name (e.g. "arduino:avr:uno").
304            verbose: Enable verbose compiler output.
305            quiet: Suppress non-error output.
306
307        Yields:
308            ``(out: str, err: str, done: bool, percent: float)`` tuples.
309
310        Raises:
311            InvalidFqbnError: If fqbn is empty.
312            CompileError: If the Compile RPC fails.
313        """
314        self._ensure_connected()
315        instance = self._ensure_instance()
316
317        if not fqbn:
318            raise exceptions.InvalidFqbnError("FQBN is required")
319
320        try:
321            request = compile_pb2.CompileRequest(
322                instance=instance,
323                fqbn=fqbn,
324                sketch_path=sketch_path,
325                verbose=verbose,
326                quiet=quiet,
327            )
328
329            percent = 0.0
330            for resp in self.stub.Compile(request, timeout=120):
331                out = resp.out_stream.decode() if resp.HasField("out_stream") else ""
332                err = resp.err_stream.decode() if resp.HasField("err_stream") else ""
333                done = resp.HasField("result") and resp.result
334                if resp.HasField("progress"):
335                    percent = resp.progress.percent
336                if done:
337                    percent = 100.0
338                yield out, err, done, percent
339
340        except grpc.RpcError as e:
341            raise exceptions.CompileError(f"Compile RPC failed: {e.details()}")
342        except Exception as e:
343            raise exceptions.CompileError(f"Compile failed: {e}")
344
345    def compile(
346        self,
347        sketch_path: str,
348        fqbn: str,
349        verbose: bool = False,
350        quiet: bool = False,
351    ) -> CompileResult:
352        """Compile an Arduino sketch.
353
354        Args:
355            sketch_path: Path to the sketch directory containing the .ino file.
356            fqbn: Fully Qualified Board Name (e.g. "arduino:avr:uno").
357            verbose: Enable verbose compiler output.
358            quiet: Suppress non-error output.
359
360        Returns:
361            CompileResult with success flag, output, and error messages.
362
363        Raises:
364            InvalidFqbnError: If fqbn is empty.
365            CompileError: If the Compile RPC fails.
366        """
367        out_lines = []
368        err_lines = []
369        success = False
370
371        for out, err, done, percent in self.compile_stream(
372            sketch_path, fqbn, verbose, quiet
373        ):
374            if out:
375                out_lines.append(out)
376            if err:
377                err_lines.append(err)
378            if done:
379                success = True
380
381        return CompileResult(
382            success=success,
383            output="".join(out_lines),
384            error="".join(err_lines),
385            sketch_path=sketch_path,
386        )

388    def upload_stream(
389        self,
390        sketch_path: str,
391        fqbn: str,
392        port: str,
393        verbose: bool = False,
394        verify: bool = True,
395    ) -> Iterator[tuple[str, str, bool]]:
396        """Stream upload output as it arrives from the gRPC stream.
397
398        Yields ``(out, err, done)`` tuples for each response from the Upload
399        RPC.  *out* and *err* are decoded text chunks (may be empty).  *done*
400        is True on the final response that contains the result.
401
402        Args:
403            sketch_path: Path to the sketch directory.
404            fqbn: Fully Qualified Board Name (e.g. "arduino:avr:uno").
405            port: Serial port address (e.g. "/dev/ttyACM0").
406            verbose: Enable verbose avrdude output.
407            verify: Verify upload after writing.
408
409        Yields:
410            ``(out: str, err: str, done: bool)`` tuples.
411
412        Raises:
413            InvalidFqbnError: If fqbn is empty.
414            InvalidPortError: If port is empty.
415            UploadError: If the Upload RPC fails.
416        """
417        self._ensure_connected()
418        instance = self._ensure_instance()
419
420        if not fqbn:
421            raise exceptions.InvalidFqbnError("FQBN is required")
422
423        if not port:
424            raise exceptions.InvalidPortError("Port is required")
425
426        try:
427            port_proto = port_pb2.Port(address=port, protocol="serial")
428
429            request = upload_pb2.UploadRequest(
430                instance=instance,
431                fqbn=fqbn,
432                sketch_path=sketch_path,
433                port=port_proto,
434                verbose=verbose,
435                verify=verify,
436            )
437
438            for resp in self.stub.Upload(request):
439                out = resp.out_stream.decode() if resp.HasField("out_stream") else ""
440                err = resp.err_stream.decode() if resp.HasField("err_stream") else ""
441                done = resp.HasField("result") and resp.result
442                yield out, err, done
443
444        except grpc.RpcError as e:
445            raise exceptions.UploadError(f"Upload RPC failed: {e.details()}")
446        except Exception as e:
447            raise exceptions.UploadError(f"Upload failed: {e}")
448
449    def upload(
450        self,
451        sketch_path: str,
452        fqbn: str,
453        port: str,
454        verbose: bool = False,
455        verify: bool = True,
456    ) -> UploadResult:
457        """Upload a compiled sketch to a board.
458
459        Args:
460            sketch_path: Path to the sketch directory.
461            fqbn: Fully Qualified Board Name (e.g. "arduino:avr:uno").
462            port: Serial port address (e.g. "/dev/ttyACM0").
463            verbose: Enable verbose avrdude output.
464            verify: Verify upload after writing.
465
466        Returns:
467            UploadResult with success flag, output, and error messages.
468
469        Raises:
470            InvalidFqbnError: If fqbn is empty.
471            InvalidPortError: If port is empty.
472            UploadError: If the Upload RPC fails.
473        """
474        out_lines = []
475        err_lines = []
476        success = False
477
478        for out, err, done in self.upload_stream(
479            sketch_path, fqbn, port, verbose, verify
480        ):
481            if out:
482                out_lines.append(out)
483            if err:
484                err_lines.append(err)
485            if done:
486                success = True
487
488        return UploadResult(
489            success=success,
490            output="".join(out_lines),
491            error="".join(err_lines),
492        )

494    def compile_and_upload(
495        self,
496        sketch_path: str,
497        fqbn: str,
498        port: str,
499        verbose: bool = False,
500        verify: bool = True,
501    ) -> tuple[CompileResult, UploadResult]:
502        """Compile a sketch then upload it to a board.
503
504        Skips upload if compilation fails.
505
506        Args:
507            sketch_path: Path to the sketch directory.
508            fqbn: Fully Qualified Board Name.
509            port: Serial port address.
510            verbose: Enable verbose output.
511            verify: Verify upload after writing.
512
513        Returns:
514            Tuple of (CompileResult, UploadResult). If compile fails, the
515            UploadResult will have success=False with an error message.
516        """
517        compile_result = self.compile(sketch_path, fqbn, verbose)
518
519        if not compile_result.success:
520            return compile_result, UploadResult(success=False, error="Compile failed")
521
522        upload_result = self.upload(sketch_path, fqbn, port, verbose, verify)
523        return compile_result, upload_result
