  92def _resolve_first_port_info(ports):
  93    """Resolve the first available board info from a ports list.
  94
  95    Args:
  96        ports: List of board info dicts.
  97
  98    Returns:
  99        Tuple of (port, fqbn, hardware_id).
 100
 101    Raises:
 102        ValueError: If no port or FQBN found.
 103    """
 104    result = get_first_board(ports)
 105    if not result:
 106        raise ValueError("port missing")
 107    port, fqbn, hw_id = result
 108    if not fqbn:
 109        raise ValueError("fqbn missing")
 110    return port, fqbn, hw_id

 113def _resolve_board_info(
 114    active_board_port, active_board_fqbn, active_board_hardware_id, ports
 115):
 116    """Resolve and reconcile board info from session state and known ports.
 117
 118    Args:
 119        active_board_port: Current active board port from session.
 120        active_board_fqbn: Current active board FQBN from session.
 121        active_board_hardware_id: Current active board hardware ID.
 122        ports: List of known port info dicts.
 123
 124    Returns:
 125        Tuple of (port, fqbn, hardware_id).
 126    """
 127    info = get_port_info(active_board_port)
 128    if not info:
 129        if active_board_fqbn:
 130            info = find_board_info_by_fqbn(active_board_fqbn, ports)
 131        if info:
 132            active_board_port = info.get("port", "")
 133            if not active_board_port:
 134                raise ValueError("port missing")
 135        elif ports:
 136            (active_board_port, active_board_fqbn, active_board_hardware_id) = (
 137                _resolve_first_port_info(ports)
 138            )
 139    else:
 140        port = info.get("port", "")
 141        if not port:
 142            raise ValueError("port missing")
 143        fqbn = info.get("fqbn", "")
 144        if not fqbn:
 145            raise ValueError("fqbn missing")
 146        if not active_board_fqbn:
 147            active_board_fqbn = fqbn
 148        elif fqbn != active_board_fqbn:
 149            info = find_board_info_by_fqbn(active_board_fqbn, ports)
 150            if info:
 151                active_board_port = info.get("port", "")
 152                if not active_board_port:
 153                    raise ValueError("port missing")
 154                active_board_fqbn = info.get("fqbn", "")
 155                if not active_board_fqbn:
 156                    raise ValueError("fqbn missing")
 157            else:
 158                active_board_fqbn = fqbn
 159    return active_board_port, active_board_fqbn, active_board_hardware_id

 162def _write_alarm_hpp(meds):
 163    """Write the alarm.hpp file from a list of medicines.
 164
 165    Args:
 166        meds: List of Medicine instances.
 167    """
 168    try:
 169        ensure_sketch_dir()
 170        path = _get_alarm_hpp_path()
 171        content = generate_alarm_hpp(meds)
 172        with open(path, "w") as f:
 173            f.write(content)
 174        logger.info("Wrote alarm.hpp (%d medicines) to %s", len(meds), path)
 175    except Exception:
 176        logger.exception("Failed to write alarm.hpp")

 209def _compute_diff(active_board):
 210    """Compute the difference between store medicines and alarm.hpp entries.
 211
 212    Args:
 213        active_board: Active board port string.
 214
 215    Returns:
 216        Diff dict with metadata, alarm_hpp, differ flag, and active_board.
 217    """
 218    metadata_list = sorted(
 219        [_medicine_to_dict(m) for m in store.only_enabled()],
 220        key=_medicine_dict_sort_key,
 221    )
 222    path = _get_alarm_hpp_path()
 223    if not os.path.exists(path):
 224        return {
 225            "metadata": metadata_list,
 226            "alarm_hpp": [],
 227            "differ": True,
 228            "alarm_hpp_exists": False,
 229            "alarm_hpp_error": None,
 230            "active_board": active_board,
 231        }
 232    try:
 233        parsed = parse_alarm_hpp(path)
 234        alarm_hpp_list = sorted(
 235            [
 236                {
 237                    "name": e["name"],
 238                    "hour": e["hour"],
 239                    "minute": e["minute"],
 240                    "day_of_week": e["day_of_week"],
 241                    "day_of_month": e["day_of_month"],
 242                }
 243                for e in parsed
 244            ],
 245            key=_medicine_dict_sort_key,
 246        )
 247        differ = metadata_list != alarm_hpp_list
 248        return {
 249            "metadata": metadata_list,
 250            "alarm_hpp": alarm_hpp_list,
 251            "differ": differ,
 252            "alarm_hpp_exists": True,
 253            "alarm_hpp_error": None,
 254            "active_board": active_board,
 255        }
 256    except Exception as e:
 257        return {
 258            "metadata": metadata_list,
 259            "alarm_hpp": [],
 260            "differ": True,
 261            "alarm_hpp_exists": True,
 262            "alarm_hpp_error": str(e),
 263            "active_board": active_board,
 264        }

 280

 282        Tuple of (success_bool, error_message).
 283    """

 662            f'<div class="log-viewer">Loaded {len(entries)} medicine(s) from alarm.hpp.</div>',

 864    @app.route("/sketch/upload", methods=["POST"])
 865    def html_sketch_upload():
 866        """Handle sketch file upload."""
 867        request.get_data()
 868        files = request.files.getlist("files[]")
 869        hardware_id = request.args.get("hardware_id", "")
 870        if not files:
 871            return _render_sketch_path_selector()
 872
 873        first_rel = files[0].filename or ""
 874        parts = first_rel.replace("\\", "/").split("/")
 875        root_name = parts[0] if parts else "sketch"
 876
 877        server_ts = datetime.datetime.utcnow().isoformat()
 878        ts_dir = datetime.datetime.utcnow().strftime("%Y%m%d_%H%M%S_%f")
 879        safe_name = secure_filename(root_name) or "sketch"
 880        ip = request.remote_addr or "unknown"
 881        ua = request.headers.get("User-Agent", "unknown")
 882        salt = hashlib.sha256(f"{ip}:{ua}".encode()).hexdigest()[:16]
 883        upload_dir = os.path.join(state.UPLOAD_BASE_DIR, f"{salt}_{ts_dir}_{safe_name}")
 884        os.makedirs(upload_dir, exist_ok=True)
 885
 886        total_size = 0
 887        file_count = 0
 888        for f in files:
 889            rel_path = f.filename or ""
 890            if not rel_path:
 891                continue
 892            file_count += 1
 893            rel_parts = rel_path.replace("\\", "/").split("/")
 894            safe_parts = [secure_filename(p) or p for p in rel_parts]
 895            safe_rel = os.path.join(*safe_parts)
 896            dest = os.path.join(upload_dir, safe_rel)
 897            os.makedirs(os.path.dirname(dest), exist_ok=True)
 898            f.save(dest)
 899            total_size += os.path.getsize(dest)
 900
 901        sketch_dir = os.path.join(upload_dir, safe_name)
 902        _normalize_ino_filename(sketch_dir, safe_name)
 903
 904        checksum = _compute_sketch_checksum(sketch_dir)
 905        key = (ip, ua)
 906
 907        with state._upload_registry_lock:
 908            user_sketches = state._upload_registry.setdefault(key, {})
 909            versions = user_sketches.setdefault(safe_name, [])
 910            existing = _find_existing_version(user_sketches, checksum)
 911            if existing:
 912                shutil.rmtree(upload_dir, ignore_errors=True)
 913                sketch_dir = existing["path"]
 914                if hardware_id and hardware_id not in existing["hardware_ids"]:
 915                    existing["hardware_ids"].append(hardware_id)
 916                    existing["board_timestamps"][hardware_id] = server_ts
 917                    _update_meta_hw_ids(
 918                        existing["path"],
 919                        existing["hardware_ids"],
 920                        existing["board_timestamps"],
 921                    )
 922                    _save_registry()
 923                    broadcast_ws(
 924                        '<div class="sketch-event">Sketch deduplicated <!-- board-event --></div>'
 925                    )
 926            else:
 927                meta = {
 928                    "ip": ip,
 929                    "user_agent": ua,
 930                    "server_timestamp": server_ts,
 931                    "file_count": len(files),
 932                    "total_size": total_size,
 933                    "root_name": safe_name,
 934                    "checksum": checksum,
 935                    "hardware_ids": [hardware_id] if hardware_id else [],
 936                    "board_timestamps": {hardware_id: server_ts} if hardware_id else {},
 937                }
 938                with open(os.path.join(upload_dir, ".meta"), "w") as mf:
 939                    json.dump(meta, mf)
 940                bisect.insort(
 941                    versions,
 942                    {
 943                        "path": sketch_dir,
 944                        "checksum": checksum,
 945                        "server_timestamp": server_ts,
 946                        "hardware_ids": meta["hardware_ids"],
 947                        "board_timestamps": meta["board_timestamps"],
 948                    },
 949                    key=lambda v: v["server_timestamp"],
 950                )
 951                _save_registry()
 952                broadcast_ws(
 953                    '<div class="sketch-event">Sketch updated <!-- board-event --></div>'
 954                )
 955
 956        if hardware_id:
 957            set_assignment(hardware_id, sketch_dir)
 958
 959        return _render_sketch_path_selector(sketch_dir, hardware_id=hardware_id)

 965        hardware_id = request.args.get("hardware_id", "")
 966        if not sketch_path:
 967            return _render_sketch_path_selector(hardware_id=hardware_id)

 970        if not norm_path.startswith(norm_base):
 971            return _render_sketch_path_selector(hardware_id=hardware_id)
 972
 973        ip = request.remote_addr or "unknown"
 974        ua = request.headers.get("User-Agent", "unknown")
 975        key = (ip, ua)
 976        removed = False
 977        with state._upload_registry_lock:
 978            entries = state._upload_registry.get(key, {})
 979            for name, versions in list(entries.items()):
 980                for i, v in enumerate(versions):
 981                    if not removed and v["path"] == norm_path:
 982                        versions.pop(i)
 983                        removed = True
 984                        if not versions:
 985                            del entries[name]
 986                        break
 987            if removed:
 988                all_latest = [vs[-1] for vs in entries.values() if vs]
 989                latest = (
 990                    max(all_latest, key=lambda v: v["server_timestamp"])
 991                    if all_latest
 992                    else None
 993                )
 994        if removed:
 995            _save_registry()
 996            broadcast_ws(
 997                '<div class="sketch-event">Sketch deleted <!-- board-event --></div>'
 998            )

1000            if latest is not None:
1001                return _render_sketch_path_selector(
1002                    latest.get("path", ""), hardware_id=hardware_id
